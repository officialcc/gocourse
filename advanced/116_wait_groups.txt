package main

import (
	"fmt"
	"sync"
	"time"
)

// === CONSTRUCTION EXAMPLE ===
type Worker struct {
	Id int
	Task string
}

// PerformTask simulates a worker performing a task
func (w *Worker) PerformTask(wg *sync.WaitGroup) {
	defer wg.Done()
	fmt.Printf("Worker ID %d started %s\n", w.Id, w.Task)
	time.Sleep(time.Second)
	fmt.Printf("Worker ID %d finished %s\n", w.Id, w.Task)
}

func main() {
	var wg sync.WaitGroup
	
	// Define tasks to be performed by workers
	tasks := []string{"digging", "laying bricks", "painting"}

	for i, task := range tasks {
		worker := Worker{Id: i+1, Task: task}
		wg.Add(1)
		go worker.PerformTask(&wg)
	}

	// Wait for all works to finish
	wg.Wait()

	// Construction is finished
	fmt.Println("Construction finished")
}
