package main

import "fmt"

func main() {
	// for used as while if omit initialize & statements
	i := 1
	for i <= 5 {
		fmt.Println("Iteration:", i)
		i++
	}

	// with break
	sum := 0
	for {
		sum += 10
		fmt.Println("Sum:", sum)

		if (sum >= 50) {
			break
		}
	}

	// with continue
	num := 1
	for num <= 10 {
		if num%2 == 0 {
			num++
			continue
		}
		fmt.Println("Odd Number:", num)
		num++ // ++ increment operator -> increases value by 1; -- decrement operator -> decreases value by 1
	}
}
