package main

import (
	"fmt"
	// "net/http"
	demo "net/http" // naming import as "demo"
)

func main() {
	fmt.Println("Hello, Go Standard Library")

	resp, err := demo.Get("https://jsonplaceholder.typicode.com/posts/1") // changed "http.Get" to "demo.Get"
	if err != nil {
		fmt.Println("Error: ", err)
		return
	}
	defer resp.Body.Close()

	fmt.Println("HTTP Response Status:", resp.Status)
}
