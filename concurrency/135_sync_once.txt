package main

import (
	"fmt"
	"sync"
)

var once sync.Once

func initialize() {
	fmt.Println("This will not be repeated no matter how many times you call this function using once.Do")
}

func main() {

	var wg sync.WaitGroup
	for i := range 5 {
		wg.Add(1)
		go func() {
			defer wg.Done()
			fmt.Println("Goroutine No.:", i)
			once.Do(initialize) // uncomment to print initialize line only once
			// initialize() // uncomment to print initialize line for every Goroutine
		}()
	}
	wg.Wait()
}
